from otel_helpers import catchtime, init_otel from opentelemetry import trace, metrics from calls import cpu_intensive_call init_otel() tracer = trace.get_tracer(__name__) meter = metrics.get_meter(__name__) execution_time_hgram = meter.create_histogram('execution_time') for x in range(3): with tracer.start_as_current_span("cpu_intensive_application") as parent: with tracer.start_as_current_span("cpu_intensive_call") as child, catchtime() as t: cpu_intensive_call(num_iterations=5000000) execution_time_hgram.record(t())
